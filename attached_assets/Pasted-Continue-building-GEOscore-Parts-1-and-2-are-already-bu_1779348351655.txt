def create_subscription(plan_id: str, 
                        email: str) -> dict:
  
  # Create or get Razorpay customer
  customer = client.customer.create({
    "name": email.split('@')[0],
    "email": email,
    "notify": {
      "sms": False,
      "email": True
    }
  })
  
  subscription = client.subscription.create({
    "plan_id": plan_id,
    "customer_notify": 1,
    "quantity": 1,
    "total_count": 12,  # 12 months
    "customer_id": customer['id'],
    "addons": [],
    "notes": {
      "email": email
    }
  })
  
  return {
    "subscription_id": subscription['id'],
    "razorpay_key": RAZORPAY_KEY_ID,
    "customer_id": customer['id']
  }

--- VERIFY PAYMENT SIGNATURE ---

def verify_payment(razorpay_payment_id: str,
                  razorpay_subscription_id: str,
                  razorpay_signature: str) -> bool:
  
  message = f"{razorpay_payment_id}|{razorpay_subscription_id}"
  
  generated_signature = hmac.new(
    RAZORPAY_KEY_SECRET.encode(),
    message.encode(),
    hashlib.sha256
  ).hexdigest()
  
  return generated_signature == razorpay_signature

async def verify_payment_endpoint(

  is_valid = razorpay_service.verify_payment(
    request.razorpay_payment_id,

@router.post("/api/payment/webhook")
async def razorpay_webhook(
  request: Request):
  
  # Verify webhook signature
  webhook_body = await request.body()
  signature = request.headers.get(
    'X-Razorpay-Signature')
  
  expected = hmac.new(
    RAZORPAY_WEBHOOK_SECRET.encode(),
    webhook_body,
    hashlib.sha256
  ).hexdigest()
  
  if signature != expected:
    raise HTTPException(400, "Invalid signature")
  
  data = await request.json()
  event = data.get('event')
  
  if event == 'subscription.activated':
    await handle_subscription_activated(data)
  elif event == 'subscription.cancelled':
    await handle_subscription_cancelled(data)
  elif event == 'payment.failed':
    await handle_payment_failed(data)
  
  return {"status": "ok"}

from apscheduler.schedulers.asyncio import (
  AsyncIOScheduler)

scheduler = AsyncIOScheduler()

--- DAILY MONITORING JOB ---
(Runs 2am UTC every day)

@scheduler.scheduled_job('cron', hour=2, minute=0)
async def daily_monitoring_job():
  
  # Get all active paid users
  users = await get_active_paid_users()
  
  for user in users:
    brands = await get_user_brands(user['id'])
    
    for brand in brands:
      try:
        # Run full audit
        keywords = await get_domain_keywords(
          brand['domain'])
        category_data = {
          'brand_name': brand['brand_name'],
          'category': brand['category'],
          'market': brand['market'],
          'top_competitors': brand['competitors']
        }
        
        prompts = generate_prompts(
          brand_name=brand['brand_name'],
          category=brand['category'],
          market=brand['market'],
          keywords=keywords,
          competitors=brand['competitors'],
          is_paid=True
        )
        
        ai_results = await run_all_queries(prompts)
        scores = calculate_total_score(
          brand['brand_name'], ai_results)
        
        # Get yesterday's score
        yesterday_score = await get_yesterday_score(
          brand['id'])
        
        # Save today's score
        await save_daily_score(
          brand_id=brand['id'],
          scores=scores
        )
        
        # Generate new recommendations 
        # if Monday
        from datetime import datetime
        if datetime.now().weekday() == 0:
          recommendations = await (
            generate_recommendations(
              brand['brand_name'],
              brand['domain'],
              brand['category'],
              brand['market'],
              scores,
              keywords
            ))
          await save_recommendations(
            brand['id'], recommendations)
          
          # Send weekly digest
          await send_weekly_digest(
            email=user['email'],
            domain=brand['domain'],
            current_score=scores['total'],
            last_score=yesterday_score,
            ai_status={
              'chatgpt': scores['chatgpt']['found'],
              'gemini': scores['gemini']['found'],
              'perplexity': 
                scores['perplexity']['found']
            },
            top_recommendations=recommendations[:3]
          )
        
        # Score drop alert 
        # (if dropped more than 10 points)
        if (yesterday_score and 
            yesterday_score - scores['total'] > 10):
          await send_score_drop_alert(
            email=user['email'],
            domain=brand['domain'],
            old_score=yesterday_score,
            new_score=scores['total']
          )
          
      except Exception as e:
        print(f"Error monitoring {brand['domain']}: 
              {e}")
        continue
