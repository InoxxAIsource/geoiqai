=== IMPLEMENTATION ===

Update the /api/agent/chat endpoint:

Current:
const response = await claude.messages.create({
  model: "claude-sonnet-4-20250514",
  system: systemPrompt,
  messages: conversationHistory
})

New (with tools):
const response = await claude.messages.create({
  model: "claude-sonnet-4-20250514",
  system: systemPrompt,
  messages: conversationHistory,
  tools: [
    runAuditTool,
    getKeywordDataTool,
    getCompetitorDataTool,
    generateGeoFileTool,
    checkTechnicalAuditTool
  ]
})

Handle tool use response:
if (response.stop_reason === 'tool_use') {
  
  const toolUseBlock = response.content
    .find(b => b.type === 'tool_use')
  
  const toolName = toolUseBlock.name
  const toolInput = toolUseBlock.input
  
  let toolResult
  
  if (toolName === 'run_audit') {
    toolResult = await runFullAudit(
      toolInput.domain)
  }
  
  if (toolName === 'get_keyword_data') {
    toolResult = await getKeywordData(
      toolInput.domain)
  }
  
  if (toolName === 'get_competitor_data') {
    toolResult = await getCompetitorData(
      toolInput.domain,
      toolInput.competitors)
  }
  
  if (toolName === 'generate_geo_file') {
    toolResult = await generateGeoFile(
      toolInput.domain,
      toolInput.file_type)
  }
  
  if (toolName === 'check_technical_audit') {
    toolResult = await checkTechnicalAudit(
      toolInput.domain)
  }
  
  // Send tool result back to Claude
  const finalResponse = await 
    claude.messages.create({
      model: "claude-sonnet-4-20250514",
      system: systemPrompt,
      messages: [
        ...conversationHistory,
        { 
          role: "assistant", 
          content: response.content 
        },
        {
          role: "user",
          content: [{
            type: "tool_result",
            tool_use_id: toolUseBlock.id,
            content: JSON.stringify(toolResult)
          }]
        }
      ],
      tools: allTools
    })
  
  return finalResponse.content[0].text
}
