Perplexity. Think of it as Google Search Console but 
for AI search systems.

=== DATABASE SCHEMA ===
Create all these tables in Supabase.
Run these SQL commands in Supabase SQL editor:

-- Users table
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  razorpay_customer_id TEXT,
  razorpay_subscription_id TEXT,
  plan TEXT DEFAULT 'free',
  created_at TIMESTAMP DEFAULT NOW(),
  last_login TIMESTAMP
);

-- Audits table
CREATE TABLE audits (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  url TEXT NOT NULL,
  domain TEXT NOT NULL,
  brand_name TEXT,
  category TEXT,
  market TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  score_total INTEGER DEFAULT 0,
  score_chatgpt INTEGER DEFAULT 0,
  score_gemini INTEGER DEFAULT 0,
  score_perplexity INTEGER DEFAULT 0,
  chatgpt_found BOOLEAN DEFAULT FALSE,
  gemini_found BOOLEAN DEFAULT FALSE,
  perplexity_found BOOLEAN DEFAULT FALSE,
  chatgpt_detail TEXT,
  gemini_detail TEXT,
  perplexity_detail TEXT,
  competitors_found JSONB DEFAULT '[]',
  keywords_used JSONB DEFAULT '[]',
  raw_results JSONB DEFAULT '{}',
  ip_address TEXT,
  is_paid BOOLEAN DEFAULT FALSE
);

-- Monitored brands table
CREATE TABLE monitored_brands (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id),
  domain TEXT NOT NULL,
  brand_name TEXT,
  category TEXT,
  market TEXT,
  keywords JSONB DEFAULT '[]',
  competitors JSONB DEFAULT '[]',
  dataforseo_keywords JSONB DEFAULT '[]',
  gsc_connected BOOLEAN DEFAULT FALSE,
  bing_connected BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW(),
  last_checked TIMESTAMP
);

-- Daily scores table
CREATE TABLE daily_scores (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  brand_id UUID REFERENCES monitored_brands(id),
  date DATE NOT NULL,
  score_total INTEGER DEFAULT 0,
  score_chatgpt INTEGER DEFAULT 0,
  score_gemini INTEGER DEFAULT 0,
  score_perplexity INTEGER DEFAULT 0,
  recommendations JSONB DEFAULT '[]',
  created_at TIMESTAMP DEFAULT NOW()
);

-- Email subscribers table
CREATE TABLE email_subscribers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  domain TEXT,
  audit_id UUID REFERENCES audits(id),
  subscribed_at TIMESTAMP DEFAULT NOW(),
  unsubscribed_at TIMESTAMP
);

-- IP rate limiting table
CREATE TABLE rate_limits (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  ip_address TEXT NOT NULL,
  audit_count INTEGER DEFAULT 1,
  date DATE DEFAULT CURRENT_DATE,
  UNIQUE(ip_address, date)
);
