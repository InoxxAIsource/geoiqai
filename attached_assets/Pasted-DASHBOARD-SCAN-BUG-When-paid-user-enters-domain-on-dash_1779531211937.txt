=== DATABASE FOR VISIBILITY TAB ===

Citations need to be extracted 
during every audit and stored.

For Perplexity API response:
Parse the citations array that 
Perplexity returns automatically.
Each citation has url and title.
Store each in citations table.

For ChatGPT and Gemini:
After main query completes,
send follow-up query:
"List only the specific website URLs 
that informed your previous answer. 
Return one URL per line, 
nothing else."
Parse URLs from response.
Store each in citations table.

CREATE TABLE IF NOT EXISTS citations (
  id UUID PRIMARY KEY 
    DEFAULT gen_random_uuid(),
  audit_id UUID REFERENCES audits(id),
  brand_id UUID,
  ai_system TEXT NOT NULL,
  prompt TEXT,
  cited_url TEXT,
  cited_domain TEXT,
  citation_type TEXT DEFAULT 'authority',
  times_cited INTEGER DEFAULT 1,
  first_seen TIMESTAMPTZ DEFAULT NOW(),
  last_seen TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS custom_prompts (
  id UUID PRIMARY KEY
    DEFAULT gen_random_uuid(),
  brand_id UUID,
  prompt_text TEXT NOT NULL,
  tags TEXT[] DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW()
);
