Razorpay webhook handler:
async function handleWebhook(event) {
  
  if (event.type === 
      'subscription.activated') {
    
    const email = event.payload
      .subscription.entity
      .notes.email
    
    const subscriptionId = 
      event.payload.subscription
      .entity.id
    
    const plan = getPlanFromAmount(
      event.payload.subscription
      .entity.plan_id
    )
    
    // Create or update user
    const user = await db.upsert('users', {
      email: email,
      razorpay_subscription_id: 
        subscriptionId,
      plan: plan,
      created_at: new Date()
    }, { onConflict: 'email' })
    
    // Generate login link
    // (magic link — no password needed)
    const loginToken = generateToken()
    await saveLoginToken(
      user.id, loginToken)
    
    // Send welcome email with login link
    await sendWelcomeEmail(
      email,
      `https://geoiqai.com/auth/
       magic?token=${loginToken}`
    )
    
    // If email was in subscribers table
    // link their audit history
    await linkSubscriberHistory(email, 
      user.id)
  }
  
  if (event.type === 
      'subscription.cancelled') {
    await db.update('users',
      { plan: 'free' },
      { email: event.payload
        .subscription.entity
        .notes.email }
    )
  }
}
