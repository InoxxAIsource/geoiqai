=== IMPLEMENTATION ===

In the agent response handler:

After Claude returns text response:
1. Analyze the conversation context
2. Determine which visual to render
3. Render visual component below text

Detection function:
function detectVisualType(
  userMessage, agentResponse) {
  
  const msg = userMessage.toLowerCase()
  
  if (msg.includes('score') || 
      msg.includes('low') ||
      msg.includes('why') ||
      msg.includes('visibility')) {
    return 'score_breakdown'
  }
  
  if (msg.includes('competitor') ||
      msg.includes('compare') ||
      msg.includes('vs') ||
      msg.includes('beating')) {
    return 'competitor_chart'
  }
  
  if (msg.includes('citation') ||
      msg.includes('listed') ||
      msg.includes('sources')) {
    return 'citation_gap'
  }
  
  if (msg.includes('do first') ||
      msg.includes('priority') ||
      msg.includes('fix') ||
      msg.includes('next step') ||
      msg.includes('what should')) {
    return 'action_cards'
  }
  
  if (msg.includes('progress') ||
      msg.includes('trend') ||
      msg.includes('history') ||
      msg.includes('over time')) {
    return 'trend_chart'
  }
  
  if (msg.includes('blog') ||
      msg.includes('article') ||
      msg.includes('write post')) {
    return 'blog_card'
  }
  
  if (msg.includes('tweet') ||
      msg.includes('twitter') ||
      msg.includes('social')) {
    return 'tweet_cards'
  }
  
  if (msg.includes('calendar') ||
      msg.includes('plan') ||
      msg.includes('schedule') ||
      msg.includes('week')) {
    return 'content_calendar'
  }
  
  if (msg.includes('keyword') ||
      msg.includes('rank for') ||
      msg.includes('search')) {
    return 'keyword_table'
  }
  
  if (msg.includes('technical') ||
      msg.includes('robots') ||
      msg.includes('schema') ||
      msg.includes('crawler')) {
    return 'technical_scorecard'
  }
  
  return null // text only response
}
