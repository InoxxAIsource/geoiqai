import jsPDF from 'jspdf'
import autoTable from 'jspdf-autotable'

async function generatePDFReport(
  brandData, auditData, competitorData) {
  
  const doc = new jsPDF()
  
  // Cover page
  doc.setFontSize(24)
  doc.setFont('helvetica', 'bold')
  doc.text('AI Visibility Report', 20, 40)
  
  doc.setFontSize(14)
  doc.setFont('helvetica', 'normal')
  doc.text(brandData.domain, 20, 55)
  doc.text(
    `Generated: ${new Date()
      .toLocaleDateString()}`, 
    20, 65)
  
  // Score section
  doc.addPage()
  doc.setFontSize(18)
  doc.text('Executive Summary', 20, 20)
  
  doc.setFontSize(36)
  doc.setTextColor(
    auditData.scoreTotal > 66 ? 
      [5, 150, 105] : // green
    auditData.scoreTotal > 33 ? 
      [217, 119, 6] : // amber
      [220, 38, 38]  // red
  )
  doc.text(
    `${auditData.scoreTotal}/100`, 
    20, 50)
  
  // AI systems table
  autoTable(doc, {
    startY: 70,
    head: [['AI System', 
            'Score', 'Status']],
    body: [
      ['ChatGPT', 
       `${auditData.scoreChatgpt}/33`,
       auditData.chatgptStatus],
      ['Gemini',
       `${auditData.scoreGemini}/33`,
       auditData.geminiStatus],
      ['Perplexity',
       `${auditData.scorePerplexity}/33`,
       auditData.perplexityStatus],
      ['Claude',
       `${auditData.scoreClaude}/33`,
       auditData.claudeStatus],
      ['Grok',
       `${auditData.scoreGrok}/33`,
       auditData.grokStatus],
    ]
  })
  
  // Technical audit page
  doc.addPage()
  doc.setFontSize(18)
  doc.setTextColor(0, 0, 0)
  doc.text('Technical GEO Audit', 20, 20)
  
  autoTable(doc, {
    startY: 30,
    head: [['Check', 'Score', 'Status']],
    body: [
      ['robots.txt',
       `${auditData.robotsScore}/100`,
       auditData.robotsStatus],
      ['llms.txt',
       `${auditData.llmsScore}/100`,
       auditData.llmsStatus],
      ['Schema Markup',
       `${auditData.schemaScore}/100`,
       auditData.schemaStatus],
      ['Content Structure',
       `${auditData.contentScore}/100`,
       auditData.contentStatus],
      ['Entity Consistency',
       `${auditData.entityScore}/100`,
       auditData.entityStatus],
    ]
  })
  
  // Fix actions page
  doc.addPage()
  doc.setFontSize(18)
  doc.text('Priority Fix Actions', 20, 20)
  
  const topActions = 
    fixActions.slice(0, 5)
  
  autoTable(doc, {
    startY: 30,
    head: [['Priority', 'Task', 
            'Impact', 'Time']],
    body: topActions.map(action => [
      action.priority,
      action.title,
      `+${action.impact} pts`,
      action.timeEstimate
    ])
  })
  
  // Save
  doc.save(
    `geoiq-report-${brandData.domain}
     -${Date.now()}.pdf`)
}
