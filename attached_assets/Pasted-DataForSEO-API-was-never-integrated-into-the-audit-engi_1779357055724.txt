import httpx
import base64
import json
from datetime import datetime, timedelta

DATAFORSEO_BASE = "https://api.dataforseo.com"

def get_auth_header():
    login = os.environ.get("DATAFORSEO_LOGIN")
    password = os.environ.get("DATAFORSEO_PASSWORD")
    credentials = f"{login}:{password}"
    encoded = base64.b64encode(
        credentials.encode()).decode()
    return {
        "Authorization": f"Basic {encoded}",
        "Content-Type": "application/json"
    }

async def get_domain_keywords(
    domain: str,
    location_code: int = 2356
) -> list:
    
    # Check database cache first
    # If cached within 7 days → return cached
    cached = await get_cached_keywords(domain)
    if cached:
        print(f"Keywords from cache: {domain}")
        return cached
    
    # Call DataForSEO API
    payload = [{
        "target": domain,
        "language_code": "en",
        "location_code": location_code,
        "limit": 20,
        "order_by": [
            "keyword_data.keyword_info"
            ".search_volume,desc"
        ],
        "filters": [
            "keyword_data.keyword_info"
            ".search_volume", ">", 100
        ]
    }]
    
    try:
        async with httpx.AsyncClient() as client:
            response = await client.post(
                f"{DATAFORSEO_BASE}/v3/"
                "dataforseo_labs/google/"
                "ranked_keywords/live",
                headers=get_auth_header(),
                json=payload,
                timeout=20
            )
        
        data = response.json()
        
        # Extract keywords
        keywords = []
        tasks = data.get("tasks", [])
        if tasks and tasks[0].get("result"):
            items = tasks[0]["result"][0].get(
                "items", [])
            for item in items[:15]:
                kw_data = item.get(
                    "keyword_data", {})
                kw_info = kw_data.get(
                    "keyword_info", {})
                keyword = kw_data.get(
                    "keyword", "")
                volume = kw_info.get(
                    "search_volume", 0)
                
                if keyword and volume > 0:
                    keywords.append({
                        "keyword": keyword,
                        "volume": volume,
                        "competition": kw_info.get(
                            "competition", 0)
                    })
        
        # Cache in database for 7 days
        if keywords:
            await cache_keywords(domain, keywords)
            print(f"DataForSEO: {len(keywords)} "
                  f"keywords for {domain}")
        else:
            print(f"DataForSEO: no keywords "
                  f"found for {domain}, "
                  f"using AI fallback")
        
        return keywords
        
    except Exception as e:
        print(f"DataForSEO error: {e}")
        return []  # Trigger AI fallback

CREATE TABLE keyword_cache (

    domain TEXT UNIQUE NOT NULL,
    keywords JSONB NOT NULL,
    location_code INTEGER DEFAULT 2356,
    cached_at TIMESTAMP DEFAULT NOW(),
    expires_at TIMESTAMP DEFAULT 
        NOW() + INTERVAL '7 days'
);

Add these two functions:

async def get_cached_keywords(
    domain: str) -> list:
    result = await db.query(
        "SELECT keywords FROM keyword_cache "
        "WHERE domain = $1 "
        "AND expires_at > NOW()",
        domain
    )
    if result:
        return result[0]["keywords"]
    return None

async def cache_keywords(
    domain: str, 
    keywords: list):
    await db.execute(
        "INSERT INTO keyword_cache "
        "(domain, keywords) VALUES ($1, $2) "
        "ON CONFLICT (domain) DO UPDATE SET "
        "keywords = $2, "
        "cached_at = NOW(), "
        "expires_at = NOW() + INTERVAL '7 days'",
        domain, json.dumps(keywords)
    )

        return 2356   # India
    elif domain.endswith(".co.uk"):
        return 2826   # UK
