DETECTION LOGIC:

function getChips(userMessage, agentResponse) {
  const msg = userMessage.toLowerCase()
  const resp = agentResponse.toLowerCase()

  // Check agent response content first
  if (resp.includes('tweet 1') || 
      resp.includes('tweet 2')) {
    return CHIPS.after_tweets_written
  }
  
  if (resp.length > 800 && 
      (resp.includes('## ') || 
       resp.includes('diabetes meal') ||
       resp.includes('breakfast for'))) {
    return CHIPS.after_blog_written
  }
  
  if (resp.includes('monday') || 
      resp.includes('tuesday') ||
      resp.includes('next 7 days')) {
    return CHIPS.after_calendar
  }

  // Then check user message intent
  if (msg.includes('score') || 
      msg.includes('low') || 
      msg.includes('why')) {
    return CHIPS.after_score
  }
  
  if (msg.includes('competitor') || 
      msg.includes('compare') || 
      msg.includes('vs')) {
    return CHIPS.after_competitor
  }
  
  if (msg.includes('citation') || 
      msg.includes('cited') || 
      msg.includes('listed')) {
    return CHIPS.after_citations
  }
  
  if (msg.includes('technical') || 
      msg.includes('robots') || 
      msg.includes('setup')) {
    return CHIPS.after_technical
  }
  
  if (msg.includes('keyword') || 
      msg.includes('target')) {
    return CHIPS.after_keywords
  }
  
  if (msg.includes('plan') || 
      msg.includes('calendar') || 
      msg.includes('week')) {
    return CHIPS.after_calendar
  }
  
  if (msg.includes('first') || 
      msg.includes('priority') || 
      msg.includes('should i')) {
    return CHIPS.after_priorities
  }

  return CHIPS.default
}
