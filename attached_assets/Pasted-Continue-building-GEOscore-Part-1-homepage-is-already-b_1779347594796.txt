def validate_and_scrape(url: str) -> dict:
  
  # Clean the URL
  if not url.startswith('http'):
    url = 'https://' + url
  
  # Extract domain
  domain = extract_domain(url)
  
  # Scrape homepage
  try:
    response = requests.get(url, timeout=8, 
      headers={'User-Agent': 
      'Mozilla/5.0 GEOscore Bot'})
    soup = BeautifulSoup(response.text, 
      'html.parser')
    
    title = soup.find('title').text if 
      soup.find('title') else ''
    meta_desc = soup.find('meta', 
      {'name': 'description'})
    meta_desc = meta_desc['content'] if 
      meta_desc else ''
    h1 = soup.find('h1').text if 
      soup.find('h1') else ''
    body_text = ' '.join(
      soup.get_text().split()[:500])
    
    return {
      'domain': domain,
      'title': title,
      'meta_description': meta_desc,
      'h1': h1,
      'body_text': body_text,
      'success': True
    }
  except:
    return {'domain': domain, 
            'success': False}

import base64
import httpx

DATAFORSEO_URL = "https://api.dataforseo.com"

def get_auth_header():
  credentials = f"{DATAFORSEO_LOGIN}:{DATAFORSEO_PASSWORD}"
  encoded = base64.b64encode(
    credentials.encode()).decode()
  return {"Authorization": f"Basic {encoded}",
          "Content-Type": "application/json"}

async def get_domain_keywords(domain: str, 
                               location: int = 2356) -> list:
  # Check Supabase cache first (7 day cache)
  cached = await check_keyword_cache(domain)
  if cached:
    return cached
  
  payload = [{
    "target": domain,
    "language_code": "en",
    "location_code": location,
    "limit": 20,
    "order_by": ["keyword_data.keyword_info.search_volume,desc"]
  }]
  
  async with httpx.AsyncClient() as client:
    response = await client.post(
      f"{DATAFORSEO_URL}/v3/dataforseo_labs/"
      f"google/ranked_keywords/live",
      headers=get_auth_header(),
      json=payload,
      timeout=15
    )
  
  data = response.json()
  
  keywords = []
  try:
    items = data['tasks'][0]['result'][0]['items']
    for item in items[:10]:
      keywords.append({
        'keyword': item['keyword_data']['keyword'],
        'volume': item['keyword_data']
          ['keyword_info']['search_volume']
      })
  except:
    keywords = []
  
  # Cache in Supabase for 7 days
  if keywords:
    await cache_keywords(domain, keywords)
  
  return keywords

async def detect_category(scraped_data: dict, 
                          keywords: list) -> dict:
  
  client = anthropic.Anthropic(
    api_key=ANTHROPIC_API_KEY)
  
  prompt = f"""Analyze this website and return 
JSON only, no other text:

Title: {scraped_data.get('title', '')}
Description: {scraped_data.get('meta_description', '')}
H1: {scraped_data.get('h1', '')}
Content: {scraped_data.get('body_text', '')[:300]}
Top keywords: {[k['keyword'] for k in keywords[:5]]}

Return this exact JSON structure:
{{
  "brand_name": "extracted brand name",
  "category": "one of: health app, saas tool, 
    ecommerce, fintech, edtech, food delivery, 
    travel, real estate, other",
  "market": "one of: India, Global, US, UK",
  "product_description": "one sentence description",
  "top_competitors": ["competitor1", 
    "competitor2", "competitor3"]
}}"""

  message = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=500,
    messages=[{"role": "user", "content": prompt}]
  )
  
  import json
  try:
    return json.loads(message.content[0].text)
  except:
    return {
      "brand_name": scraped_data['domain'],
      "category": "saas tool",
      "market": "India",
      "product_description": "",
      "top_competitors": []
    }

--- STEP 4: PROMPT GENERATION ---

def generate_prompts(brand_name: str, 
                    category: str,
                    market: str, 
                    keywords: list,
                    competitors: list,
                    is_paid: bool) -> list:
  
  prompts = []
  
  # Brand prompts (always)
  brand_prompts = [
    f"What is {brand_name}?",
    f"{brand_name} review",
    f"Is {brand_name} good?",
    f"{brand_name} alternatives"
  ]
  prompts.extend(brand_prompts[:3 
    if not is_paid else 4])
  
  # Keyword prompts from DataForSEO
  if keywords:
    for kw in keywords[:5 if not is_paid else 10]:
      prompts.append(kw['keyword'])
  else:
    # Fallback prompts
    fallback = [
      f"best {category} for {market}",
      f"top {category} apps {market}",
      f"{category} recommendations {market}"
    ]
    prompts.extend(fallback[:2 
      if not is_paid else 3])
  
  # Competitor prompts (paid only)
  if is_paid and competitors:
    for comp in competitors[:2]:
      prompts.append(
        f"alternatives to {comp}")
      prompts.append(
        f"{brand_name} vs {comp}")
  
  return prompts

async def query_chatgpt(prompt: str) -> str:
  client = openai.AsyncOpenAI(
    api_key=OPENAI_API_KEY)
  
  response = await client.chat.completions.create(
    model="gpt-4o-mini",
    temperature=0.3,
    max_tokens=500,
    messages=[
      {"role": "system", 
       "content": "You are a helpful assistant. "
       "Answer questions directly and recommend "
       "specific products or services where relevant."},
      {"role": "user", "content": prompt}
    ]
  )
  return response.choices[0].message.content

async def query_gemini(prompt: str) -> str:
  import google.generativeai as genai
  genai.configure(api_key=GEMINI_API_KEY)
  model = genai.GenerativeModel('gemini-1.5-flash')
  response = model.generate_content(prompt)
  return response.text

async def query_perplexity(prompt: str) -> str:
  async with httpx.AsyncClient() as client:
    response = await client.post(
      "https://api.perplexity.ai/chat/completions",
      headers={
        "Authorization": f"Bearer {PERPLEXITY_API_KEY}",
        "Content-Type": "application/json"
      },
      json={
        "model": "llama-3.1-sonar-small-128k-online",
        "messages": [
          {"role": "system", 
           "content": "Be precise and recommend "
           "specific products where relevant."},
          {"role": "user", "content": prompt}
        ],
        "max_tokens": 500,
        "temperature": 0.3
      },
      timeout=15
    )
    data = response.json()
    return data['choices'][0]['message']['content']

async def run_all_queries(prompts: list) -> dict:
  results = {
    'chatgpt': [],
    'gemini': [],
    'perplexity': []
  }
  
  for prompt in prompts:
    tasks = [
      query_chatgpt(prompt),
      query_gemini(prompt),
      query_perplexity(prompt)
    ]
    
    responses = await asyncio.gather(
      *tasks, return_exceptions=True)
    
    results['chatgpt'].append({
      'prompt': prompt,
      'response': responses[0] 
        if not isinstance(responses[0], Exception) 
        else 'timeout'
    })
    results['gemini'].append({
      'prompt': prompt,
      'response': responses[1] 
        if not isinstance(responses[1], Exception) 
        else 'timeout'
    })
    results['perplexity'].append({
      'prompt': prompt,
      'response': responses[2] 
        if not isinstance(responses[2], Exception) 
        else 'timeout'
    })
  
  return results

def calculate_score(brand_name: str, 
                   ai_responses: list) -> dict:
  
  brand_lower = brand_name.lower()
  score = 0
  found = False
  detail = ""
  competitor_mentions = []
  
  for item in ai_responses:
    response = item.get('response', '').lower()
    
    if brand_lower in response:
      found = True
      
      # Position bonus
      first_mention = response.find(brand_lower)
      total_length = len(response)
      
      if first_mention < total_length * 0.3:
        score += 15  # Found early = strong signal
      else:
        score += 10  # Found but buried
      
      # Extract detail for UI
      sentences = response.split('.')
      for s in sentences:
        if brand_lower in s:
          detail = s.strip()[:120]
          break
  
  # Cap at 33 per system
  score = min(score, 33)
  
  return {
    'found': found,
    'score': score,
    'detail': detail if found 
      else "Not found in AI responses",
    'competitor_mentions': competitor_mentions
  }

def calculate_total_score(brand_name: str, 
                          all_results: dict) -> dict:
  
  chatgpt_score = calculate_score(
    brand_name, all_results['chatgpt'])
  gemini_score = calculate_score(
    brand_name, all_results['gemini'])
  perplexity_score = calculate_score(
    brand_name, all_results['perplexity'])
  
  total = (chatgpt_score['score'] + 
           gemini_score['score'] + 
           perplexity_score['score'])
  
  return {
    'total': min(total, 100),
    'chatgpt': chatgpt_score,
    'gemini': gemini_score,
    'perplexity': perplexity_score
  }

@router.post("/api/audit")
async def run_audit(request: AuditRequest, 
                   req: Request):
  
  # Rate limiting
  ip = req.client.host
  if not await check_rate_limit(ip):
    raise HTTPException(429, 
      "Free limit reached (3/day). "
      "Upgrade for unlimited checks.")
  
  # Check 24h cache
  cached = await check_audit_cache(
    request.domain)
  if cached:
    cached['from_cache'] = True
    return cached
  
  # Run full pipeline
  scraped = await validate_and_scrape(request.url)
  keywords = await get_domain_keywords(
    scraped['domain'])
  category_data = await detect_category(
    scraped, keywords)
  
  prompts = generate_prompts(
    brand_name=category_data['brand_name'],
    category=category_data['category'],
    market=category_data['market'],
    keywords=keywords,
    competitors=category_data['top_competitors'],
    is_paid=False
  )
  
  ai_results = await run_all_queries(prompts)
  scores = calculate_total_score(
    category_data['brand_name'], ai_results)
  
  # Save to Supabase
  audit_id = await save_audit(
    scraped, category_data, 
    keywords, scores, ai_results, ip)
  
  return {
    'audit_id': audit_id,
    'domain': scraped['domain'],
    'brand_name': category_data['brand_name'],
    'category': category_data['category'],
    'market': category_data['market'],
    'scores': scores,
    'from_cache': False
  }
